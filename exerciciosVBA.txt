Option Explicit

Sub exercicio01()
  Dim num1 As Double, num2 As Double, resultado As Double
  num1 = InputBox("Digite um numero")
  num2 = InputBox("Digite um numero")

End Sub

Sub exercicio03()
  Dim resp As String
  resp = InputBox("Digite seu gênero: (F - feminino, M - masculino)")
  Select Case resp
    Case "F", "f"
      MsgBox "Feminino"
    Case "M", "m"
      MsgBox "Masculino"
    Case Else
      MsgBox "Sexo não mencionado"
    End Select
End Sub

Sub exercicio11()
  Dim num1 As Double, num2 As Double, resultado As Double, operacao As String
  num1 = InputBox("Digite um numero")
  num2 = InputBox("Digite outro numero")
  operacao = InputBox("Digite a operação (+, -, * ou /)")
  Select Case operacao
    Case "+"
      resultado = num1 + num2
      MsgBox "Resultado da soma: " & resultado
      If resultado < 0 Then
        MsgBox "Negativo"
      Else
        MsgBox "Positivo"
      End If
      If resultado Mod 2 = 0 Then
        MsgBox "Par"
      Else
        MsgBox "Negativo"
      End If
    Case "/"
      resultado = num1 / num2
      MsgBox "Resultado da divisão: " & resultado
      If resultado < 0 Then
        MsgBox "Negativo"
      Else
        MsgBox "Positivo"
      End If
      If resultado Mod 2 = 0 Then
        MsgBox "Par"
      Else
        MsgBox "Negativo"
      End If
    Case "-"
      resultado = num1 - num2
      MsgBox "Resultado da subtração: " & resultado
      If resultado < 0 Then
        MsgBox "Negativo"
      Else
        MsgBox "Positivo"
      End If
      If resultado Mod 2 = 0 Then
        MsgBox "Par"
      Else
        MsgBox "Negativo"
      End If
    Case "*"
      resultado = num1 * num2
      MsgBox "Resultado da multiplicação: " & resultado
      If resultado < 0 Then
        MsgBox "Negativo"
      Else
        MsgBox "Positivo"
      End If
      If resultado Mod 2 = 0 Then
        MsgBox "Par"
      Else
        MsgBox "Negativo"
      End If
    End Select
End Sub
